def <=> (otro)
	if(@autor == otro.get_autor)
		if(@publicacion == otro.get_publicacion)
			if(@titulo == otro.get_titulo)
				return 0
			else
				arr = [@titulo, otro.get_titulo]
				arr.sort_by!{|t| t.downcase}
				if(arr.first == @titulo)
					return 1
				end
				return -1
			end
		elsif publicacion > otro.get_publicacion
			return -1
		else
			return 1
		end
	else
		arr = [@autor, otro.get_autor]
		arr.sort_by!{|t| t.downcase}
		if(arr.first == @autor)
			return -1
		end
		return 1
	end
end
