def insertar_inicio(val)
        if @inicio.nil?
                @inicio = Struct::Nodo.new(nil, val, nil)
                @final = @inicio
        else
                copia = @inicio
                @inicio = Struct::Nodo.new(nil, val, copia)
                copia[:ant] = @inicio
        end
end

def tamano()
        if !@inicio.nil?

                contador = 1
                copia = @inicio

                while !copia[:sig].nil?
                        contador += 1
                        copia2 = copia[:sig]
                        copia = copia2
                end
        end
        return contador
end

def to_s
