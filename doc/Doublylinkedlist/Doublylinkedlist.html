def insertar_inicio(val)
	if @inicio.nil?
		@inicio = Struct::Nodo.new(nil, val, nil)
		@final = @inicio
	else
		copia = @inicio
		@inicio = Struct::Nodo.new(nil, val, copia)
		copia[:ant] = @inicio
	end
end

def ordenar! 
	cambio = true
	while cambio
		cambio = false
		i = @inicio
		i_1 = @inicio[:sig]
		while i_1 != nil
			if(i[:valor] > i_1[:valor])
				i[:valor], i_1[:valor] = i_1[:valor], i[:valor]
				cambio = true
			end
			i = i_1
			i_1 = i_1[:sig]
		end
	end
end

def tamano()
	if !@inicio.nil?

		contador = 1
		copia = @inicio

		while !copia[:sig].nil?
			contador += 1
			copia2 = copia[:sig]
			copia = copia2
		end
	end
	return contador
end

def to_s
	actual = @inicio
	cadena = "|"
		while !actual.nil?
			cadena << actual[:valor].to_s

			if !actual[:sig].nil?
				cadena << ", "
			end

			actual = actual[:sig]
		end
	cadena << "|"
	return cadena
end
